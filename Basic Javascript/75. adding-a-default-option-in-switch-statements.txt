function switchOfStuff(val) {
  let answer = "";
  // Only change code below this line

    case "a": return "apple";
    break;
    case "b": return "bird";
    break;
    case "c": return "cat";
    break;
    default:return "stuff";
    break;
  }
  // Only change code above this line
  return answer;
}
